296 static inline uint32_t SOC_REG_TO_ULP_PERIPH_SEL(uint32_t reg) {
297  uint32_t ret = 3;
298  if (reg < DR_REG_RTCCNTL_BASE) {
299  assert(0 && "invalid register base");
300  } else if (reg < DR_REG_RTCIO_BASE) {
301  ret = RD_REG_PERIPH_RTC_CNTL;
302  } else if (reg < DR_REG_SENS_BASE) {
303  ret = RD_REG_PERIPH_RTC_IO;
304  } else if (reg < DR_REG_RTC_I2C_BASE){
305  ret = RD_REG_PERIPH_SENS;
306  } else if (reg < DR_REG_IO_MUX_BASE){
307  ret = RD_REG_PERIPH_RTC_I2C;
308  } else {
309  assert(0 && "invalid register base");
310  }
311  return ret;
312 }
